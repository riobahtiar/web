## Code Examples with Syntax Highlighting

### JavaScript Example

\`\`\`javascript
// Calculate fibonacci numbers
function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

console.log(fibonacci(10)); // 55
\`\`\`

### TypeScript Example

\`\`\`typescript
interface BlogPost {
  title: string;
  description: string;
  created_at: Date;
  tags: string[];
}

const post: BlogPost = {
  title: "My Post",
  description: "A great post",
  created_at: new Date(),
  tags: ["typescript", "astro"]
};
\`\`\`
